    @media (prefers-reduced-motion: reduce) {

     * Scroll-cue bob — 2s ease-in-out infinite; 0→4px→0 translateY.
     * Exposed as .animate-scrollbob utility and wrapped in a no-preference
     * media query so it fully stops under reduced motion (not just paused).
     * Per 02-UI-SPEC §Hero "Scroll cue visual".
     */
    @keyframes scrollbob {
      0%,
      100% {
        transform: translateY(0);
      }
      50% {
        transform: translateY(4px);
      }
    }

    @media (prefers-reduced-motion: no-preference) {
      .animate-scrollbob {
        animation: scrollbob 2s ease-in-out infinite;
      }
    }

    /*
     * Typing caret blink — 1.0s infinite opacity 1→0→1.
     * Consumed by src/components/HeroTyping.tsx (Plan 02) on the resting caret.
     * Per 02-UI-SPEC §Typing Animation "Caret".
     * Note: reduced-motion is handled INSIDE HeroTyping.tsx (the caret is not

      0%,

      }
      50% {
        opacity: 0;
      }
    }

    }
