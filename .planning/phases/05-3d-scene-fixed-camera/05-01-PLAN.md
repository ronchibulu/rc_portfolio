**Main component:**
```tsx
export default function GameSetupScene() {
  const { scene, cameras } = useGLTF('/models/gaming_setup_v12.glb');
  const { camera, invalidate } = useThree();

  // After model resolves: point camera at target + render the first demand frame + signal ready.
  useEffect(() => {
    camera.lookAt(new THREE.Vector3(...CAMERA_TARGET));
    invalidate();
    $sceneReady.set(true);
  }, [camera, invalidate]);

  return (
    <>
      {/* Camera — makeDefault registers it as the View's active camera. */}
      <PerspectiveCamera
        makeDefault
        position={CAMERA_POSITION}
        fov={CAMERA_FOV}
        near={CAMERA_NEAR}
        far={CAMERA_FAR}
      />

      {/* Lighting rig — matches dark purple atmosphere in image.png (UI-SPEC §2). */}
      {/* No env map, no shadows — too heavy for demand frameloop. */}
      <ambientLight intensity={0.15} color="#1a0a2e" />
      <directionalLight position={[5, 8, 5]}   intensity={1.2} color="#b8a0ff" castShadow={false} />
      <directionalLight position={[-3, 4, -5]} intensity={0.4} color="#6040ff" castShadow={false} />

      {/* Model — desk_spot is the root node. */}
      <primitive object={scene} />
    </>
  );
}
```

**SceneView.tsx full content:**
```tsx
import { View }               from '@react-three/drei/web/View.js';
import { Suspense, useRef }   from 'react';
import GameSetupScene,
       { SceneLoader }        from '@/components/canvas/GameSetupScene';

interface SceneViewProps {
  trackId: string;
}

export default function SceneView({ trackId }: SceneViewProps) {
  // This component is client:only="react" — document is always available.
  // Initialize ref synchronously from the Astro-rendered DOM element.
  const ref = useRef<HTMLElement | null>(
    typeof document !== 'undefined'
      ? document.getElementById(trackId)
      : null
  );

  return (
    <View track={ref as React.MutableRefObject<HTMLElement>}>
      <Suspense fallback={<SceneLoader />}>
        <GameSetupScene />
      </Suspense>
    </View>
  );
}
```
