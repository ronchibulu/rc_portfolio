### Task 3 — Add ScrollTrigger tagline animation component (ScrollNarrative.tsx)

Create `src/components/canvas/ScrollNarrative.tsx` — a React island that owns the tagline animation and hard-cut transition logic:

```tsx
import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { useGSAP } from '@gsap/react';
import { $sceneReady } from '@/stores';
import { useStore } from '@nanostores/react';

gsap.registerPlugin(ScrollTrigger, useGSAP);

export default function ScrollNarrative() {
  const sceneReady = useStore($sceneReady);

  useGSAP(() => {
    if (!sceneReady) return;
    if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return;

    const tl = gsap.timeline({
      scrollTrigger: {
        trigger: '#scene-scroll-pin',
        start: 'top top',
        end: 'bottom top',
        scrub: true,
      },
    });

    // Tagline 1: fade in 20%-50%, fade out 70%-90%
    tl.fromTo('#fly-tagline-1', { opacity: 0 }, { opacity: 1, ease: 'power1.in' }, 0.2)
      .to('#fly-tagline-1', { opacity: 0, ease: 'power1.out' }, 0.7);

    // Tagline 2: fade in 30%-60%, fade out 75%-90%
    tl.fromTo('#fly-tagline-2', { opacity: 0 }, { opacity: 1, ease: 'power1.in' }, 0.3)
      .to('#fly-tagline-2', { opacity: 0, ease: 'power1.out' }, 0.75);

    // Hard cut: at scroll end (progress = 1), reveal OS screen section
    ScrollTrigger.create({
      trigger: '#scene-scroll-pin',
      start: 'bottom top',
      onEnter: () => {
        // Instantly hide taglines (already at 0 from timeline end)
        // OS screen (#projects section) is already below in DOM — scroll arrives there naturally
      },
      onLeaveBack: () => {
        // User scrolled back up — taglines will re-animate via scrub timeline
      },
    });

  }, { dependencies: [sceneReady] });

  return null; // No DOM output — pure scroll orchestration
}
```

Mount in `index.astro` with `client:only="react"` immediately after `<SceneCanvas>`.

### Task 6 — Reduced-motion static fallback

When `prefers-reduced-motion: reduce`:
- `ScrollNarrative` returns early without creating any ScrollTrigger
- Camera stays at `CAMERA_POSITION` (no useFrame lerp since `cameraProgress.value` stays 0)
- Taglines remain `opacity-0` (CSS hidden, not announced)
- User scrolls naturally to the projects section

Add CSS in `globals.css` for reduced-motion tagline visibility:

```css
@media (prefers-reduced-motion: reduce) {
  #fly-tagline-1,
  #fly-tagline-2 {
    display: none; /* skip entirely, they're aria-hidden decorative */
  }
}
```
